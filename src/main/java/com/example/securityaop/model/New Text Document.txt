package com.example.securityaop.model;

import jakarta.persistence.*;
import java.time.LocalDateTime;

/**
 * SecurityLog
 * ----------------------------
 * Stores important system and user-level security actions.
 * Each entry records who performed the action, when, and what it was.
 */
@Entity
@Table(name = "security_logs")
public class SecurityLog {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    private String email;        // Who performed the action
    private String action;       // What happened (e.g., "Deleted user@example.com")
    private String type;         // Type: INFO, ADMIN_ACTION, WARNING, ERROR
    private LocalDateTime timestamp; // When it occurred

    public Long getId() { return id; }
    public void setId(Long id) { this.id = id; }

    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }

    public String getAction() { return action; }
    public void setAction(String action) { this.action = action; }

    public String getType() { return type; }
    public void setType(String type) { this.type = type; }

    public LocalDateTime getTimestamp() { return timestamp; }
    public void setTimestamp(LocalDateTime timestamp) { this.timestamp = timestamp; }
}
